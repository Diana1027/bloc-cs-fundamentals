Programming Assignment: 

1. You and a friend have set a wager to see who can find the word "Albatross" in the dictionary the fastest. Write a program to allow you to win the bet.

A: var words = ["Albatross", "Cat", "Dog", "Frog", "Mouse"]

function findWord(arr, word){
  let low = arr[0];
  let high = arr[arr.length - 1];

  while(low <= high){
    let mid = (arr.indexOf(low) + arr.indexOf(high)) / 2;
    mid = Math.round(mid);

    if(arr[mid] > word){
      high = arr[mid - 1];
    } else if(arr[mid] < word){
      low = arr[mid + 1];
    } else {
      return arr[mid]
    }
  }
  return "Not Found";
}
findWord(words, "Albatross");


2. You work at a pet store, and a child has asked you to net the only white-spotted goldfish from the fish tank. Write a program that will help you net the right fish.

var fishTank = ["goldfish", "shark", "seahorse", "turtle", "white-spotted goldfish"];

function getFish(arr, fish){
  for(var i = 0; i < arr.length; i++){
    if(arr[i] === fish){
      return "You've caught a " + fish
    }
  }
  return "That fish isn't in the fish tank"
}
getFish(fishTank, "white-spotted goldfish");
